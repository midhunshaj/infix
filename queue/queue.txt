#include <stdio.h> 
#include <stdlib.h> 
#define QUEUE_SIZE 5 
int q[QUEUE_SIZE], rear = -1, front = 0; 
void enqueue(int item) 
{ 
    if (rear == QUEUE_SIZE - 1) 
    { 
        printf(" Queue is overflow "); 
    } 
    else 
    { 
        rear = rear + 1; 
        q[rear] = item; 
    } 
} 
 
void dequeue() 
{ 
    int item; 
    if (front > rear) 
    { 
        printf(" Queue is underflow "); 
    } 
    else 
    { 
        item = q[front]; 
        front = front + 1; 
        printf(" Element %d is deleted ", item); 
    } 
} 
 
void display() 
{ 
    int i; 
    if (front > rear) 
    { 
        printf(" Queue is empty "); 
    } 
    else 
    { 
        printf(" Elements in queue are :- \n"); 
        for (i = front; i <= rear; i++) 
        { 
            printf("%d", q[i]); 
        } 
    } 
} 
 
int main() 
{ 
    int ch, item; 
    for (;;) 
    { 
        printf("\n QUEUE OPERATIONS \n"); 
        printf("1. Enqueue\n2. Dequeue\n3. Display\n4. Exit\n"); 
        printf("Enter a choice [1 to 4]:- "); 
        scanf("%d", &ch); 
        switch (ch) 
        { 
            case 1: 
                printf(" Enter element to insert:- "); 
                scanf("%d", &item); 
                enqueue(item); 
                break; 
            case 2: 
                dequeue(); 
                break; 
            case 3: 
                display(); 
                break; 
            case 4: 
                printf(" Exit "); 
                exit(0); 
        } 
    } 
}
